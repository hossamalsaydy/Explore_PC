        _basketItems.add(product);
        _productQuantities[product.productId!] = 1;
        notifyListeners();
    }
}
  1. التحقق من عدم وجود المنتج مسبقًا في السلة

void updateQuantity(int productId, int newQuantity) {
    if (_productQuantities.containsKey(productId)) {
        _productQuantities[productId] = newQuantity;
        notifyListeners();
    }
}

كيف يعمل:

  • التحقق من وجود المنتج: containsKey() يتأكد من وجود المنتج في السلة
  • تحديث القيمة: _productQuantities[productId] = newQuantity يعدل الكمية المخزنة
  • التحديث الفوري: notifyListeners() ينشط إعادة بناء الواجهة

            final response = await http.get(Uri.parse(LinkProducts));
            
            if (response.statusCode == 200) {

                    .map((item) => Computer.fromJson(item))
                    .toList();
            } else {
                throw Exception('Failed to load products');

public function getMerchantOrders($merchantId) {
    $orders = Order::whereHas('items.product', function($query) use ($merchantId) {
            $query->where('MerchantId', $merchantId);
        })
        ->with(['customer', 'items.product'])
        ->get()
        ->map(function ($order) {
            return [
                'orderId' => $order->orderId,
                'customerName' => $order->customer->UserName,
                'items' => $order->items->map(function ($item) {
                    return [
                        'productName' => $item->product->name,
                        'quantity' => $item->quantity
                    ];
                })
            ];
        });
    
    return $orders;
}

public function sendVerificationCode(Request $request) {
    $request->validate(['email' => 'required|email|exists:customs']);
    
    $code = Str::random(6);
    
    DB::table('password_reset_tokens')->updateOrInsert(
        ['email' => $request->email],
        ['token' => $code, 'created_at' => now()]
    );
    
    Mail::to($request->email)->send(new VerificationCodeMail($code));
    
    return response()->json(['message' => 'تم الإرسال']);
}

public function resetPassword(Request $request) {
    $record = DB::table('password_reset_tokens')
        ->where('email', $request->email)
        ->where('token', $request->code)
        ->first();
    
    if (!$record) {
        return response()->json(['error' => 'الرمز غير صحيح'], 400);
    }
    
    Custom::where('email', $request->email)
        ->update(['UserPassword' => $request->new_password]);
    
    DB::table('password_reset_tokens')->where('email', $request->email)->delete();
    
    return response()->json(['message' => 'تم التغيير بنجاح']);
}
